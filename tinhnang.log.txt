4. CẤU TRÚC DATABASE
4.1. Tables trong Module Package
sql
-- packages table
CREATE TABLE packages (
    id BIGINT PRIMARY KEY,
    name VARCHAR(255),
    type ENUM('member', 'agent', 'agency'),
    role_name VARCHAR(100) UNIQUE,
    price DECIMAL(10,2),
    credits_per_month INT,
    max_agents INT NULL,
    is_active BOOLEAN DEFAULT true,
    sort_order INT,
    limits JSON, -- {listingsPerMonth, vipListings, teamMembers, apiCalls, storage}
    features JSON, -- [{code, enabled}]
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- user_packages table
CREATE TABLE user_packages (
    id BIGINT PRIMARY KEY,
    user_id BIGINT FOREIGN KEY,
    package_id BIGINT FOREIGN KEY,
    status ENUM('active', 'expired', 'cancelled'),
    credits_remaining INT,
    listings_used_this_month INT,
    bonus_credits INT DEFAULT 0,
    last_reset_at TIMESTAMP,
    started_at TIMESTAMP,
    expires_at TIMESTAMP,
    cancelled_at TIMESTAMP NULL,
    created_at TIMESTAMP,
    updated_at TIMESTAMP
);

-- credit_transactions table
CREATE TABLE credit_transactions (
    id BIGINT PRIMARY KEY,
    user_id BIGINT FOREIGN KEY,
    user_package_id BIGINT FOREIGN KEY,
    type ENUM('purchase', 'usage', 'refund', 'bonus'),
    amount INT,
    balance_after INT,
    description VARCHAR(255),
    reference_type VARCHAR(50), -- property, project, api_call
    reference_id BIGINT,
    metadata JSON,
    created_at TIMESTAMP
);

-- usage_logs table
CREATE TABLE usage_logs (
    id BIGINT PRIMARY KEY,
    user_id BIGINT FOREIGN KEY,
    user_package_id BIGINT FOREIGN KEY,
    resource_type VARCHAR(50), -- listing, vip_listing, api_call
    quantity INT,
    deducted_credits BOOLEAN,
    credits_used INT,
    reference_type VARCHAR(50),
    reference_id BIGINT,
    metadata JSON,
    created_at TIMESTAMP
);

8. LUỒNG XỬ LÝ KHI USER NÂNG CẤP GÓI
text
┌─────────────────────────────────────────────────────────────────┐
│                    USER NÂNG CẤP GÓI                            │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│  1. User chọn gói muốn nâng cấp                                 │
│     POST /api/v1/packages/purchase                              │
│     { package_id: 2, payment_method: 'vnpay' }                  │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│  2. PackageService xử lý                                        │
│     - Deactivate package cũ                                     │
│     - Tạo UserPackage mới (status = 'pending')                  │
│     - Gọi Payment module để xử lý thanh toán                    │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│  3. Thanh toán thành công                                       │
│     - Update UserPackage: status = 'active'                     │
│     - started_at = now()                                        │
│     - expires_at = now()->addMonth()                            │
│     - credits_remaining = package.credits_per_month            │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│  4. Gán permissions cho user                                    │
│     - Xóa role cũ                                               │
│     - Gán role mới: $user->assignRole($package->role_name)     │
│     - Sync permissions từ package                               │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│  5. Ghi nhận giao dịch credits                                  │
│     - Tạo CreditTransaction: type = 'purchase'                  │
│     - amount = package.credits_per_month                        │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│  6. Kết quả trả về cho user                                     │
│     {                                                           │
│       success: true,                                            │
│       package: {...},                                           │
│       permissions: [...],                                       │
│       credits: 30,                                              │
│       expires_at: '2026-04-15'                                  │
│     }                                                           │
└─────────────────────────────────────────────────────────────────┘

    $user = auth()->user();
    
    // Kiểm tra hạn mức
    $remaining = Property::getRemainingQuota($user->id);
    
    if ($remaining <= 0) {
        return response()->json([
            'error' => 'Bạn đã hết hạn mức đăng tin tháng này',
            'upgrade_suggestions' => PackageService::getUpgradeSuggestions($user->id)
        ], 403);
    }
    
    // Kiểm tra nếu là VIP
    if ($request->is_vip) {
        if (!$user->can('listing.vip')) {
            return response()->json([
                'error' => 'Gói hiện tại không hỗ trợ tin VIP'
            ], 403);
        }
        
        // Kiểm tra còn VIP quota không
        $vipUsed = $user->properties()
            ->where('is_vip', true)
            ->whereMonth('created_at', now()->month)
            ->count();
            
        $package = $user->activePackage;
        $vipLimit = $package ? $package->package->limits['vipListings'] : 0;
        
        if ($vipUsed >= $vipLimit) {
            return response()->json([
                'error' => 'Bạn đã hết số lượng tin VIP trong tháng'
            ], 403);
        }
    }
    
    // ... tiếp tục xử lý
}
9.2. Reset hàng tháng (Cron job)
php
// App\Console\Kernel.php
protected function schedule(Schedule $schedule)
{
    $schedule->call(function () {
        PackageService::resetMonthlyUsage();
    })->monthly();
}

// PackageService.php
public static function resetMonthlyUsage()
{
    UserPackage::where('status', 'active')
        ->chunk(100, function($packages) {
            foreach ($packages as $package) {
                $package->update([
                    'listings_used_this_month' => 0,
                    'last_reset_at' => now(),
                    'credits_remaining' => $package->package->credits_per_month
                ]);
            }
        });
}
